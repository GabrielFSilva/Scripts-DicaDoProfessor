/// @function Wrap

/// @param min

// Caso o valor ultrapasse o limite máximo, 
// ele será "movido" para o limite mínimo.
// Caso o valor seja reduzido abaixo do limite mínimo, 
// ele será "movido" para o limite máximo.

		if (argument0 > argument2)
			argument0 += argument1 - argument2 - 1;
		else if (argument0 < argument1)
